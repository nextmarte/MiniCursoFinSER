  mutate(faixa_inferior = if_else(value < (mean_value - sd_value), value, as.numeric(NA)),

  geom_point(aes(x = Data, y = faixa_superior), color = "green") +

  geom_hline(yintercept = (mean_value + sd_value), color = "purple", linetype = "dotted") +
